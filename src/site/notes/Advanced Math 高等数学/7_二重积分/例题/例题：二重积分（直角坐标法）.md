$$
\begin{align}
m  & = min(f(x,y)) \\
M  & = max(f(x,y)) \\
 \\
\int_{D}  \int mg(x,y)dxdy \leq \int_{D}  &  \int f(x,y)g(x,y)dxdy \leq  \int_{D}  \int Mg(x,y)dxdy \\
